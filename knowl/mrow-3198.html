\begin{align*} f'(x)&=x^x\left[1+\log x\right] \\ \amp\implies \color{red}{f'(1)}\color{red}{=1}\\ f''(x)&=\diff{}{x}\left\{x^x\right\}[1+\log x] + x^x \diff{}{x}\left\{1+\log x\right\}\\ &=\left(x^x\left[1+\log x\right]\right)\left[1+\log x\right]+x^x\cdot\frac{1}{x}\\ &=x^x\left((1+\log x)^2+\frac{1}{x}\right)\\ \amp\implies\color{red}{f''(1)}\color{red}{=2}\\ f'''(x)&=\diff{}{x}\left\{x^x\right\}\left((1+\log x)^2+\frac{1}{x}\right)+x^x\diff{}{x}\left\{(1+\log x)^2+\frac{1}{x}\right\}\\ &=x^x\left[1+\log x\right]\left(\left(1+\log x\right)^2+\frac{1}{x}\right)+x^x \left[\frac{2}{x}(1+\log x)-\frac{1}{x^2}\right]\\ &=x^x\left(\left(1+\log x\right)^3+\frac{3}{x}\left(1+\log x\right)-\frac{1}{x^2}\right)\\ \amp\implies\color{red}{f'''(1)}\color{red}{=3} \end{align*}
