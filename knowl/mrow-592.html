\begin{align*} \left| \frac{1}{g(x)} - \frac{1}{G} \right| &= \left| \frac{G - g(x)}{g(x) \cdot G} \right|\\ &= |g(x) - G| \cdot \frac{1}{|G|\cdot |g(x)|} & \text{ by assumption}\\ & \lt \frac{\epsilon_1}{|G| \cdot |g(x)|} & \text{ sub in $\epsilon_1$}\\ & = \epsilon \cdot \frac{|G|}{2 |g(x)|} & \text{ since $\left|\frac{G}{2g(x)}\right| \lt 1$}\\ & \lt \epsilon \end{align*}
