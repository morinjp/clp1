\begin{align*} |f(x) - L| \lt \epsilon & \text{ whenever } 0 \lt |x-a| \lt \delta \end{align*}

\begin{gather*} \text{if } 0 \lt |x-a| \lt \delta \text{ then } |f(x) - L| \lt \epsilon. \end{gather*}
