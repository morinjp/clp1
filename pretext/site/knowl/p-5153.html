\begin{align*} F(x) &= \frac{3}{6}x^6 - \frac{7}{3}x^3 + \frac{2}{2}x^2 + 3x +\log|x| - \frac{1}{-1} x^{-1} + c & \text{clean it up}\\ &= \frac{1}{2}x^6 - \frac{7}{3}x^3 + x^2 + 3x +\log|x| + x^{-1} + c \end{align*}
